git version
git init     // inicia um repositório
tree .git/   // mostra a arvore

echo a > a.txt //cria um arquivo a.txt com o conteudo a

git commit  // passa as informações do staging area para o repositório -m "passa info"

git status  //mostra a situação o working area

git add "a.txt" // adiciona um arquivo ao staging area -a ou --all add tudo

git log //mosra a lista de commits ja realizados --online --decorate

git branch feature1 //feature é o nome da branch ele cria uma branch 

HEAD <--- é onde vc está trabalhando

git checkout feature1 // muda a HEAD para a feature1

git merge feature1  // merge entre e branch e o master quando se esta no master

git checkout -b feature2 // cria o branch e ja vai para ele

git merge feature2 // junta a feature2 ao master, quando se esta em master... 

git branch -d feature2 // deleta o branch feature2

git stash save "arquivo"

git stash list // traz os stashs  

git stash apply stash@{0} // para trazer de volta o que vc salvou no stash

git stash drop stash@{0} // dropa o stash

git stash drop stash@{0} // traz e ja dropa

git stash branch feature5

git remote add origin https://github.com/leonardoRampazzo/gitteste.git // add rep online

git push origin master //joga o master lá no rep

git pull origin master //pucha os dados do master que está lá nesta bomba, git pull ja faz merge automaticamente por isso que da muita merda

git fetch  // faz o pull pela metade não faz o merge

git diff master origin/master // ve as diferenças entre origin e master 

git merge origin/master //merge disso aqui


desfazendo merdas

git reset "hash do commit" // volta para aquele commit

git reset --soft "hash" // volta e não tira os arquivos do staging area

git reset --mixed "hash" //tira os arquivos do staging area

git reset --hard "hash" //esse arranca tudo arranca mesmo... fica do jeito que puxou ou que era no momento daquele comit


git reflog // guarda o log do git

git push origin master -f ou --force // para voltar o commmit errado no origin
